Arima vs Acp - Plots

Code 
for (i in 1:20) {
  arr_arima <- paste0("ARR", i, "_arima")
  arr_acp <- paste0("ARR", i, "_acp")
    df_long <- data.frame(
    date = as.Date(paste0(merged_data_ordered$date, "-01"), format = "%Y_%m-%d"),
    value = c(merged_data_ordered[[arr_arima]], merged_data_ordered[[arr_acp]]),
    method = rep(c("ARIMA", "ACP"), each = nrow(merged_data_ordered))
  )
    p <- ggplot(df_long, aes(x = date, y = value, color = method, group = method)) +
    geom_line() +
    labs(title = paste("Evolution des prix - ARR", i),
         x = "Date", y = "Valeur", color = "Method") +
    theme_bw() +
    theme(axis.text.x = element_text(angle = 90, hjust = 1)) +
    scale_x_date(date_labels = "%b %Y", date_breaks = "2 months", expand = c(0, 0)) +
    scale_color_manual(values = c("blue", "red"))
    print(p)
}
